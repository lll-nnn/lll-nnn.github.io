Js原型链污染

本文最后更新于：1 天前

js前置知识

javascript是一个面向对象的语言到但是并没有class和对象的区分，class只是javascript中的一个关键字。

js中对象可以是一个构造函数或者类声明

function A(a, b){
    this.aa = a
    this.bb = b
}
//也可以写成
class A {
    constructor(a, b) {
        this.aa = a
        this.bb = b
    }
}

通过new生成实例

var a = new A(1,2)
console.log(a.aa)//1
console.log(a.bb)//2

new一次就会生成一个新的实例，它们之间没有任何关系，没有共享的属性或者方法

js中的继承是通过原型链实现的

prototype，构造函数的prototype属性包含了一个对象，所有实例需要共享的属性或者方法放到这里。

function A(a, b){
    this.aa = a
    this.bb = b
}
A.prototype = {s:'s'};
var a1 = new A(1,2)
var a2 = new A(3,4)
console.log(a1.s)	//s
console.log(a2.s)	//s
A.prototype.s = 'S'
console.log(a1.s)	//S
console.log(a2.s)	//S

这样也就实现了继承属性

添加方法

function A(a, b){
    this.aa = a
    this.bb = b
}
A.prototype = {s:'s'};
A.prototype.print = function(){
    console.log("hello")
}
var a1 = new A(1,2)
var a2 = new A(3,4)
a1.print()	//hello
a2.print()	//hello

a1,a2中都没有print()这个方法，但它们有__proto__属性指向A.prototype

console.log(a1.__proto__)	//{ s: 's', print: [Function (anonymous)] }
console.log(a2.__proto__ == A.prototype)	//true

如果A.prototype中也没有print()这个方法，会继续找A.prototype.__proto__，不断向上寻找，直到null

console.log(a1.__proto__)		//{ s: 's', print: [Function (anonymous)] }
console.log(a1.__proto__.__proto__)		//[Object: null prototype] {}
console.log(a1.__proto__.__proto__.__proto__)	//null

原型链污染

> var a = {cc:3}
undefined
> a.__proto__
[Object: null prototype] {}	//a得原型是{}
> a.__proto__.cc = 9	//在a得原型上添加cc属性
9
> a1 = {}
{}
> a1.cc		//新建得a1也会有cc属性
9
> console.log(a,cc)
{ cc: 3 } 9
undefined
>

那么，在一个应用中，如果攻击者控制并修改了一个对象的原型，那么将可以影响所有和这个对象来自同一个类、父祖类的对象。这种攻击方式就是原型链污染。

什么时候可以设置__proto__的值，找可控的数组或对象的键名

var express = require('express');

const undefsafe = require('undefsafe');

var app = express();
class Notes {
    constructor() {
        this.owner = "whoknows";
        this.num = 0;
        this.note_list = {};
    }

    write_note(author, raw_note) {
        this.note_list[(this.num++).toString()] = {"author": author,"raw_note":raw_note};
    }

    get_note(id) {
        var r = {}
        undefsafe(r, id, undefsafe(this.note_list, id));
        return r;
    }

    edit_note(id, author, raw) {
        undefsafe(this.note_list, id + '.author', author);
        undefsafe(this.note_list, id + '.raw_note', raw);
    }

    get_all_notes() {
        return this.note_list;
    }

    remove_note(id) {
        delete this.note_list[id];
    }
}

var notes = new Notes();
notes.write_note("nobody", "this is nobody's first note");
