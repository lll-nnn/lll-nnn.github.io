    highlight_file(__FILE__);

    $host = $_GET['host'];
    $host = escapeshellarg($host);      //将给字符串增加一个单引号并且能引用或者转码任何已经存在的单引号
    $host = escapeshellcmd($host);        //对字符串中可能会欺骗 shell 命令执行任意命令的字符进行转义，在一些字符前加上斜杠\
    $sandbox = md5("glzjin". $_SERVER['REMOTE_ADDR']);
    echo 'you are in sandbox '.$sandbox;
    @mkdir($sandbox);       //新建一个以$sandbox命名的目录，@是压制错误提示的意思
    chdir($sandbox);       //将PHP当前目录改为$sandbox
    echo system("nmap -T5 -sT -Pn --host-timeout 2 -F ".$host);
}

谈谈escapeshellarg参数绕过和注入的问题

传入的参数是:127.0.0.1' -v -d a=1
由于escapeshellarg先对单引号转义，再用单引号将左右两部分括起来从而起到连接的作用。所以处理之后的效果:'127.0.0.1'\'' -v -d a=1'
经过escapeshellcmd针对第二步处理之后的参数中的\以及a=1’中的单引号进行处理转义之后的效果:'127.0.0.1'\\'' -v -d a=1\'
